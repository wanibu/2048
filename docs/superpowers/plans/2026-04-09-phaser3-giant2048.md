```typescript
import Phaser from 'phaser';
import { GRID_COLS, GRID_ROWS, CELL_SIZE, GRID_OFFSET_X, GRID_OFFSET_Y } from '../config';

export class Grid {
  private scene: Phaser.Scene;
  private cells: Phaser.GameObjects.Rectangle[][] = [];
  private container: Phaser.GameObjects.Container;

  // Data layer: 0 = empty, positive = block value, -1 = stone
  public data: number[][] = [];

  constructor(scene: Phaser.Scene) {
    this.scene = scene;
    this.container = scene.add.container(0, 0);
    this.initData();
    this.drawGrid();
  }

  private initData(): void {
    for (let row = 0; row < GRID_ROWS; row++) {
      this.data[row] = [];
      for (let col = 0; col < GRID_COLS; col++) {
        this.data[row][col] = 0;
      }
    }
  }

  private drawGrid(): void {
    for (let row = 0; row < GRID_ROWS; row++) {
      this.cells[row] = [];
      for (let col = 0; col < GRID_COLS; col++) {
        const { x, y } = this.cellToPixel(row, col);
        // Cell background
        const cell = this.scene.add.rectangle(x, y, CELL_SIZE - 4, CELL_SIZE - 4, 0x2a2a4a, 0.6);
        cell.setStrokeStyle(2, 0x444477);
        this.cells[row][col] = cell;
        this.container.add(cell);
      }
    }
  }

  /** Convert grid coordinates (row, col) to pixel position (center of cell) */
  cellToPixel(row: number, col: number): { x: number; y: number } {
    return {
      x: GRID_OFFSET_X + col * CELL_SIZE + CELL_SIZE / 2,
      y: GRID_OFFSET_Y + row * CELL_SIZE + CELL_SIZE / 2,
    };
  }

  /** Find the nearest grid cell for a pixel position. Returns null if out of bounds. */
  pixelToCell(px: number, py: number): { row: number; col: number } | null {
    const col = Math.round((px - GRID_OFFSET_X - CELL_SIZE / 2) / CELL_SIZE);
    const row = Math.round((py - GRID_OFFSET_Y - CELL_SIZE / 2) / CELL_SIZE);
    if (row < 0 || row >= GRID_ROWS || col < 0 || col >= GRID_COLS) return null;
    return { row, col };
  }

  /** Get the X pixel position for a column (center of column) */
  colToX(col: number): number {
    return GRID_OFFSET_X + col * CELL_SIZE + CELL_SIZE / 2;
  }

  /** Get the pixel Y of the bottom edge of the grid */
  getBottomY(): number {
    return GRID_OFFSET_Y + GRID_ROWS * CELL_SIZE;
  }

  /** Get the pixel Y of the top edge of the grid */
  getTopY(): number {
    return GRID_OFFSET_Y;
  }

  getContainer(): Phaser.GameObjects.Container {
    return this.container;
  }
}
```

```typescript
import Phaser from 'phaser';
import { CELL_SIZE, BLOCK_COLORS, valueToFrame } from '../config';

export class Shape extends Phaser.GameObjects.Container {
  public value: number;
  private bg: Phaser.GameObjects.Rectangle;
  private label: Phaser.GameObjects.Text;

  constructor(scene: Phaser.Scene, x: number, y: number, value: number) {
    super(scene, x, y);
    this.value = value;

    const size = CELL_SIZE - 8;
    const color = BLOCK_COLORS[valueToFrame(value)] ?? 0xffffff;

    this.bg = scene.add.rectangle(0, 0, size, size, color);
    this.bg.setStrokeStyle(2, 0xffffff);
    this.add(this.bg);

    this.label = scene.add.text(0, 0, String(value), {
      fontSize: '28px',
      color: '#ffffff',
      fontStyle: 'bold',
    }).setOrigin(0.5);
    this.add(this.label);

    scene.add.existing(this);
    scene.physics.add.existing(this);

    // Set physics body size
    const body = this.body as Phaser.Physics.Arcade.Body;
    body.setSize(size, size);
    body.setOffset(-size / 2, -size / 2);

    this.setSize(size, size);
  }

  getBody(): Phaser.Physics.Arcade.Body {
    return this.body as Phaser.Physics.Arcade.Body;
  }
}
```

```typescript
import Phaser from 'phaser';
import { GAME_WIDTH, GRID_COLS, CELL_SIZE, GRID_OFFSET_X, SPAWN_NUMBER_MAX, SHOOT_SPEED } from '../config';
import { Grid } from './Grid';
import { Shape } from './Shape';

export class Sling {
  private scene: Phaser.Scene;
  private grid: Grid;
  private currentShape: Shape | null = null;
  private selectedCol: number = 2; // middle column
  private columnIndicator: Phaser.GameObjects.Rectangle;
  private shootAvailable: boolean = true;
  private onShootCallback: ((shape: Shape, col: number) => void) | null = null;

  constructor(scene: Phaser.Scene, grid: Grid) {
    this.scene = scene;
    this.grid = grid;

    // Column selection indicator
    const indicatorX = grid.colToX(this.selectedCol);
    const launchY = grid.getBottomY() + 100;
    this.columnIndicator = scene.add.rectangle(indicatorX, launchY, CELL_SIZE - 4, 8, 0xffe57f, 0.6);

    this.spawnNextShape();
    this.setupInput();
  }

  private spawnNextShape(): void {
    const value = Math.pow(2, Phaser.Math.Between(1, SPAWN_NUMBER_MAX));
    const launchY = this.grid.getBottomY() + 100;
    const x = this.grid.colToX(this.selectedCol);
    this.currentShape = new Shape(this.scene, x, launchY, value);
    this.shootAvailable = true;
  }

  private setupInput(): void {
    // Keyboard: left/right to select column, space/up to shoot
    this.scene.input.keyboard!.on('keydown-LEFT', () => this.moveColumn(-1));
    this.scene.input.keyboard!.on('keydown-RIGHT', () => this.moveColumn(1));
    this.scene.input.keyboard!.on('keydown-SPACE', () => this.shoot());
    this.scene.input.keyboard!.on('keydown-UP', () => this.shoot());

    // Touch/mouse: click on grid area to select column and shoot
    this.scene.input.on('pointerdown', (pointer: Phaser.Input.Pointer) => {
      const col = Math.floor((pointer.x - GRID_OFFSET_X) / CELL_SIZE);
      if (col >= 0 && col < GRID_COLS) {
        this.selectedCol = col;
        this.updatePosition();
        this.shoot();
      }
    });
  }

  private moveColumn(delta: number): void {
    this.selectedCol = Phaser.Math.Clamp(this.selectedCol + delta, 0, GRID_COLS - 1);
    this.updatePosition();
  }

  private updatePosition(): void {
    const x = this.grid.colToX(this.selectedCol);
    this.columnIndicator.setX(x);
    if (this.currentShape) {
      this.currentShape.setX(x);
    }
  }

  private shoot(): void {
    if (!this.shootAvailable || !this.currentShape) return;
    this.shootAvailable = false;

    const shape = this.currentShape;
    this.currentShape = null;

    // Launch upward
    const body = shape.getBody();
    body.setVelocityY(-SHOOT_SPEED);

    if (this.onShootCallback) {
      this.onShootCallback(shape, this.selectedCol);
    }
  }

  onShoot(callback: (shape: Shape, col: number) => void): void {
    this.onShootCallback = callback;
  }

  respawn(): void {
    this.scene.time.delayedCall(300, () => this.spawnNextShape());
  }

  getSelectedCol(): number {
    return this.selectedCol;
  }
}
```

```typescript
import Phaser from 'phaser';
import { GAME_WIDTH, GRID_ROWS } from '../config';
import { Grid } from '../objects/Grid';
import { Sling } from '../objects/Sling';
import { Shape } from '../objects/Shape';

export class GameScene extends Phaser.Scene {
  private grid!: Grid;
  private sling!: Sling;
  private activeShapes: Shape[] = [];

  constructor() {
    super({ key: 'GameScene' });
  }

  create(): void {
    this.cameras.main.setBackgroundColor('#1a1a2e');

    // Score
    this.add.text(GAME_WIDTH / 2, 30, 'Score: 0', {
      fontSize: '24px',
      color: '#ffffff',
    }).setOrigin(0.5);

    // Grid
    this.grid = new Grid(this);

    // Sling
    this.sling = new Sling(this, this.grid);
    this.sling.onShoot((shape, col) => this.handleShoot(shape, col));
  }

  private handleShoot(shape: Shape, col: number): void {
    this.activeShapes.push(shape);
  }

  update(_time: number, _delta: number): void {
    // Check if any active shape has reached the grid or another block
    for (let i = this.activeShapes.length - 1; i >= 0; i--) {
      const shape = this.activeShapes[i];
      if (!shape.active) {
        this.activeShapes.splice(i, 1);
        continue;
      }

      const body = shape.getBody();
      const gridCell = this.grid.pixelToCell(shape.x, shape.y);

      if (!gridCell) {
        // Above grid top — let it keep going, but if it goes off screen, destroy
        if (shape.y < -50) {
          shape.destroy();
          this.activeShapes.splice(i, 1);
          this.sling.respawn();
        }
        continue;
      }

      const { row, col } = gridCell;

      // Check: has it reached a row that is occupied, or the top row?
      let shouldLand = false;

      if (row === 0) {
        // Reached top of grid
        shouldLand = true;
      } else if (this.grid.data[row][col] !== 0) {
        // This cell is occupied — land in the cell below
        shouldLand = true;
      } else {
        // Check if cell above in the direction of travel is occupied
        // Since shape moves upward, check if the row below is occupied
        // Actually, shape moves UP, so we check if we've passed into an occupied zone
        // Land if the cell directly above (row-1) or at edges
        for (let r = row; r >= 0; r--) {
          if (this.grid.data[r][col] !== 0) {
            shouldLand = true;
            break;
          }
        }
        // If no block above, check if we've reached row 0 level
        if (!shouldLand && shape.y <= this.grid.cellToPixel(0, col).y) {
          shouldLand = true;
        }
      }

      if (shouldLand) {
        this.landShape(shape, col);
        this.activeShapes.splice(i, 1);
      }
    }
  }

  private landShape(shape: Shape, col: number): void {
    // Find the lowest empty row in this column
    let targetRow = -1;
    for (let r = GRID_ROWS - 1; r >= 0; r--) {
      if (this.grid.data[r][col] === 0) {
        targetRow = r;
        break;
      }
    }

    if (targetRow === -1) {
      // Column is full — game over condition (TODO: implement in later task)
      shape.destroy();
      this.sling.respawn();
      return;
    }

    // Stop physics
    const body = shape.getBody();
    body.setVelocity(0, 0);

    // Snap to grid position
    const { x, y } = this.grid.cellToPixel(targetRow, col);
    shape.setPosition(x, y);

    // Update data layer
    this.grid.data[targetRow][col] = shape.value;

    // Tween snap animation
    this.tweens.add({
      targets: shape,
      scaleX: 1.1,
      scaleY: 1.1,
      duration: 80,
      yoyo: true,
      ease: 'Quad.easeOut',
    });

    // Respawn next shape
    this.sling.respawn();
  }
}
```

```typescript
import { Grid } from '../objects/Grid';
import { GRID_ROWS, GRID_COLS } from '../config';

interface MergeGroup {
  cells: { row: number; col: number }[];
  value: number;
}

export class MergeSystem {
  private grid: Grid;

  constructor(grid: Grid) {
    this.grid = grid;
  }

  /**
   * Scan the grid for adjacent cells with the same value.
   * Returns groups of 2+ adjacent same-value cells.
   */
  findMergeGroups(): MergeGroup[] {
    const visited: boolean[][] = [];
    for (let r = 0; r < GRID_ROWS; r++) {
      visited[r] = new Array(GRID_COLS).fill(false);
    }

    const groups: MergeGroup[] = [];

    for (let r = 0; r < GRID_ROWS; r++) {
      for (let c = 0; c < GRID_COLS; c++) {
        if (visited[r][c] || this.grid.data[r][c] <= 0) continue;

        const value = this.grid.data[r][c];
        const cells: { row: number; col: number }[] = [];
        this.bfs(r, c, value, visited, cells);

        if (cells.length >= 2) {
          groups.push({ cells, value });
        }
      }
    }

    return groups;
  }

  private bfs(
    startRow: number, startCol: number, value: number,
    visited: boolean[][], result: { row: number; col: number }[]
  ): void {
    const queue = [{ row: startRow, col: startCol }];
    visited[startRow][startCol] = true;

    while (queue.length > 0) {
      const { row, col } = queue.shift()!;
      result.push({ row, col });

      const neighbors = [
        { row: row - 1, col },
        { row: row + 1, col },
        { row, col: col - 1 },
        { row, col: col + 1 },
      ];

      for (const n of neighbors) {
        if (
          n.row >= 0 && n.row < GRID_ROWS &&
          n.col >= 0 && n.col < GRID_COLS &&
          !visited[n.row][n.col] &&
          this.grid.data[n.row][n.col] === value
        ) {
          visited[n.row][n.col] = true;
          queue.push(n);
        }
      }
    }
  }

  /**
   * Execute a merge group: remove all cells, place merged result at the
   * cell with the most neighbors (priority merge target).
   * Returns the new value and target position.
   */
  executeMerge(group: MergeGroup): { row: number; col: number; newValue: number } {
    const newValue = group.value * 2;

    // Find the cell with the most same-value neighbors (priority target)
    let bestCell = group.cells[0];
    let bestNeighborCount = 0;

    for (const cell of group.cells) {
      let count = 0;
      for (const other of group.cells) {
        if (
          Math.abs(cell.row - other.row) + Math.abs(cell.col - other.col) === 1
        ) {
          count++;
        }
      }
      if (count > bestNeighborCount) {
        bestNeighborCount = count;
        bestCell = cell;
      }
    }

    // Remove all cells in the group
    for (const cell of group.cells) {
      this.grid.removeBorder(cell.row, cell.col);
    }

    // Place merged result
    this.grid.placeBorder(bestCell.row, bestCell.col, newValue);

    return { row: bestCell.row, col: bestCell.col, newValue };
  }
}
```
